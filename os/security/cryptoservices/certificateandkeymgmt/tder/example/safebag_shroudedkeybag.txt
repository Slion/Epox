// SafeContents
// PARAM1: salt
// PARAM2: iteration count
// PARAM3: cipher_oid
// PARAM4: cipher
// PARAM5: key
// PARAM6: iv
// PARAM7: The filename of the DER encoded keyfile
// PARAM8: The name of the certificate file
// PARAM9: The 'friendly' name of the certificate
// SafeBag
SEQUENCE
	OID=PKCS12_BAGTYPES_PKCS8SHROUDEDKEYBAG
	EXPLICIT
		// KeyBag
		SEQUENCE
			SEQUENCE
				OID=$3
				SEQUENCE
					OCTETSTRING
						RAW=$1
					END
					INTEGER=$2
				END
			END
			OCTETSTRING
				ENCRYPT=$4,$5,$6
					INCLUDE_BINARY_FILE=$7
				END	
			END
		END
	END
	// Attributes
	SET
		SEQUENCE
			OID=PKCS9_LOCAL_KEYID
			SET				
				OCTETSTRING
					HASH=sha1
						INCLUDE_BINARY_FILE=$8
					END
				END
			END
		END 
		SEQUENCE
			OID=PKCS9_FRIENDLY_NAME
			SET
				BMPSTRING=$9
			END
		END	
	END
END
