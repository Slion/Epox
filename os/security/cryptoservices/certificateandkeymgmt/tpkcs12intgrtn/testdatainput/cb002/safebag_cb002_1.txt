// PARAM2: Friendly name for CRL bag

		OID=PKCS12_BAGTYPES_CRLBAG
		EXPLICIT=0
			// CRLBag
			SEQUENCE
				OID=X509CRL
				EXPLICIT=0
					OCTETSTRING
						INCLUDE_BINARY_FILE=../root5crl.der

				END
			END
		END

		SET
			SEQUENCE
				OID=PKCS9_FRIENDLY_NAME

					OUTPUT_BINARY_FILE=cb002_friendlyname
						BMPSTRING=symbian
					END
				END
			END	
		END
	END
END
