// SafeContents
// PARAM1: salt
// PARAM2: iteration count
// PARAM3: cipher_oid
// PARAM4: cipher
// PARAM5: key
// PARAM6: iv
// PARAM7: The filename of the DER encoded keyfile
// PARAM8: The name of the certificate file
// PARAM9: The 'friendly' name of the certificate
// SafeBag

//  key bag

SEQUENCE
	OID=PKCS12_BAGTYPES_KEYBAG
	EXPLICIT
		INCLUDE_BINARY_FILE=$7
	END
	// Attributes
	SET
		SEQUENCE
			OID=PKCS9_LOCAL_KEYID
			SET
				OUTPUT_BINARY_FILE=cb006_localkeyid_secondkey.der,1				
					OCTETSTRING
						HASH=sha1
							INCLUDE_BINARY_FILE=$8
						END
					END
				END
			END
		END 
		SEQUENCE
			OID=PKCS9_FRIENDLY_NAME
			SET
				OUTPUT_BINARY_FILE=cb006_friendlyname_secondkey.der,1
					BMPSTRING=symbian
				END
			END
		END	
	END
END

// Shrouded Key bag
SEQUENCE
	OID=PKCS12_BAGTYPES_PKCS8SHROUDEDKEYBAG
	EXPLICIT
		OUTPUT_BINARY_FILE=safebag_shrouded_cb006_2.der,1
			// KeyBag
			SEQUENCE
				SEQUENCE
					OID=$3
					SEQUENCE
						OCTETSTRING
							RAW=$1
						END
						INTEGER=$2
					END
				END
				OCTETSTRING
					ENCRYPT=$4,$5,$6
						INCLUDE_BINARY_FILE=$7
					END	
				END
			END
		END
	END
	// Attributes
	SET
		SEQUENCE
			OID=PKCS9_LOCAL_KEYID
			SET
				OUTPUT_BINARY_FILE=cb006_localkeyid_secondshrouded.der,1

						END
					END
				END
			END
		END

	END
