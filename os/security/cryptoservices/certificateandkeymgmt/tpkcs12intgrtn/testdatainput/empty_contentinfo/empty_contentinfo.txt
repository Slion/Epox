// A simple PKCS#12 file containing a certificate and a private key.

SEQUENCE
	// Version
	INTEGER=3

	// authSafe - PFX
	SEQUENCE
		OID=PKCS7_DATA
		EXPLICIT		
			OCTETSTRING
				OUTPUT_BINARY_FILE=authsafecontents.der,1
					// First PDU
					SEQUENCE
						INCLUDE=contentinfo.txt
					END
				END
			END
		END
	END

	// macData
	SEQUENCE
		// DigestInfo
		SEQUENCE
			SEQUENCE
				OID=SHA1
				NULL
			END
			// DIGEST		
			OCTETSTRING
				OUTPUT_BINARY_FILE=empty_contentinfo_mac
					HMAC=SHA1,d9c9d4803178c611251ccefef608bda2bf7ec183
						INCLUDE_BINARY_FILE=authsafecontents.der
					END
				END
			END		
		END		
		// macSalt
		OCTETSTRING
			OUTPUT_BINARY_FILE=empty_contentinfo_salt
				RAW=6F:85:5D:15:3C:61:8D:4C
			END
		END
		// iterations
		INTEGER=2048	
	END
END
